In this problem, the task was to check whether the word neuroverse was contained in the given string or not. This can be done by iterating over all characters in the string, and check if the current character is the potential starting character for the word neuroverse. If we encounter such a character, then we print YES along with the index of that character in the string. Otherwise, if we do not find such a character even after scanning the entire string, then we print NO. The solution could be implemented either by manually writing the code for matching the substring starting from the current character, or by using various built in libraries in different programming languages. 

#include <math.h>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <limits.h>
#include <stdbool.h>                        
int min(int x,int y){
    if(x < y) return x;
    return y;
}                    
bool check(char str[], int idx){
    bool done = true;
    char nv[] = "neuroverse";
    for(int i = idx; i < min(idx+10,strlen(str)); i++){
        if(nv[i-idx] != str[i]){
            done = false;
            break;
        }
    }
    if(strlen(str)-idx < 10) done = false;
        return done;
    }                
int main(){
    char str[50];
    scanf("%s",str);
    for(int i = 0; i < strlen(str); i++){
        if(check(str,i)){
            printf("YES\n%d\n",i);
            return 0;
        }
    }
    printf("NO\n");
    return 0;
}
